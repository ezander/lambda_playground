# Church booleans
true := λx y. x
false := λx y. y
if pred then else := pred then else

# Boolean operators/junctors (can be defined generally)
not p := if p false true
and p q := if p q p
or p q := if p p q
implies p q := if p q true
iff p q := if p q (not q)
nand p q := not (and p q)
nor p q := not (or p q)
xor p q := not (iff p q)
